```
    @charset "UTF-8";

    /**
    * @desc File Info
    * @author liqinuo

    */

    /* Module A */
    .mod_a {}

    /* module A logo */
    .mod_a_logo {}

    /* module A nav */
    .mod_a_nav {}


    /* Module B */
    .mod_b {}

    /* module B logo */
    .mod_b_logo {}

    /* module B nav */
    .mod_b_nav {}
```
